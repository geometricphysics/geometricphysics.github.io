Common Derivatives

$ \begin{eqnarray} \frac{d f}{d x} & = & \frac{d f}{d u} \frac{d u}{d x} \end{eqnarray} $

$ \begin{eqnarray} \frac{d}{d x} f & = & \lim_{h \rightarrow 0} \frac{f(u(x+h))-f(u(x))}{h} \\ & = & \lim_{h \rightarrow 0} \frac{f(u(x+h)-u(x)+u(x))-f(u(x))}{h} \\ & = & \lim_{h \rightarrow 0} \frac{f(h u'(x)+u(x))-f(u(x))}{h} \\ & = & \lim_{h \rightarrow 0} \frac{f(u(x) + h u'(x))-f(u(x))}{h u'(x)} u'(x) \\ & = & \lim_{k \rightarrow 0} \frac{f(u + k)-f(u)}{k} u'(x) \\ & = & \frac{d f}{d u} \frac{d u}{d x} \\ \end{eqnarray} $

$ \begin{eqnarray} \frac{d}{d x} (x^{n}) & = & n x^{n-1} \end{eqnarray} $

$ \begin{eqnarray} \frac{d}{d x} (x^{n}) & = & \lim_{h \rightarrow 0} \frac{(x+h)^n - x^n}{h} \\ & = & \lim_{h \rightarrow 0} \frac{x^n + n x^{n-1} h + O(h^2) - x^n}{h} \\ & = & n x^{n-1} + \lim_{h \rightarrow 0} \frac{O(h^2)}{h} \\ & = & n x^{n-1} \\ \end{eqnarray} $

$ \begin{eqnarray} \frac{d}{d x} (e^{x}) & = & e^x \end{eqnarray} $

This is the definition of $e^x$.

$ \begin{eqnarray} \frac{d}{d x} (\ln \, x) & = & \frac{1}{x} \end{eqnarray} $

$ \begin{eqnarray} \frac{d}{d x} (\ln \, x) & = & \lim_{h \rightarrow 0} \frac{\ln(x+h)-\ln(x)}{h} \\ & = & \lim_{h \rightarrow 0} \frac{\ln(1+\frac{h}{x})}{h} \\ & = & \lim_{h \rightarrow 0} \frac{x}{h}\frac{\ln(1+\frac{h}{x})}{x} \\ & = & \lim_{h \rightarrow 0} \frac{\ln(1+\frac{h}{x})^{\frac{x}{h}}}{x} \\ & = & \frac{1}{x} \lim_{n \rightarrow \infty} \ln(1+\frac{1}{n})^{n} \\ & = & \frac{1}{x} \ln \lim_{n \rightarrow \infty} (1+\frac{1}{n})^{n} \\ & = & \frac{1}{x} \ln e \\ & = & \frac{1}{x} \\ \end{eqnarray} $

$ \begin{eqnarray} \frac{d}{d x} (\sin x) & = & \cos x \end{eqnarray} $

$ \begin{eqnarray} \frac{d}{d x} (\sin x) & = & \lim_{h \rightarrow 0} \frac{\sin(x+h)-\sin(x)}{h} \\ & = & \lim_{h \rightarrow 0} \frac{\sin(x) \cos(h) + \cos(x) \sin(h) -\sin(x)}{h} \\ & = & \lim_{h \rightarrow 0} \frac{\sin(x) + \cos(x) h -\sin(x)}{h} \\ & = & \cos x \\ \end{eqnarray} $

$ \begin{eqnarray} \frac{d}{d x} (\cos x) & = & - \sin x \end{eqnarray} $

$ \begin{eqnarray} \frac{d}{d x} (\cos x) & = & \lim_{h \rightarrow 0} \frac{\cos(x+h)-\cos(x)}{h} \\ & = & \lim_{h \rightarrow 0} \frac{\cos(x) \cos(h) - \sin(x) \sin(h) -\cos(x)}{h} \\ & = & \lim_{h \rightarrow 0} \frac{\cos(x) - \sin(x) h -\cos(x)}{h} \\ & = & - \sin x \\ \end{eqnarray} $
