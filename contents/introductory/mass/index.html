Mass and Acceleration

from browser import * from three import * from workbench import * from math import pow space3D = CartesianSpace() workbench3D = Workbench(space3D.renderer, space3D.camera) giant = SphereBuilder().color("red").radius(0.2).build() giant.position = VectorE3(0.0, 4.0, 0.0) giant.mass = ScalarE3(1.0) giant.momentum = VectorE3(0.0, 0.0, 0.0) * giant.mass space3D.add(giant) dwarf = SphereBuilder().color("blue").radius(0.2).build() dwarf.position = VectorE3(0.0, -2.0, 0.0) dwarf.mass = ScalarE3(1.0) dwarf.momentum = -giant.momentum space3D.add(dwarf) dt = 0.1 k = 0.1 def setUp(): workbench3D.setUp() def tick(t): r = dwarf.position - giant.position F = k * r * (r.magnitude() - 4.0) / r.magnitude() giant.momentum = giant.momentum + F * dt dwarf.momentum = dwarf.momentum - F * dt for body in [giant, dwarf]: body.position += (body.momentum / body.mass) * dt space3D.render() def terminate(t): return t > 60 def tearDown(): workbench3D.tearDown() WindowAnimationRunner(tick, terminate, setUp, tearDown).start()
